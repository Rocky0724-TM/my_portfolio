      // Scroll Indicator
      window.onscroll = function () {
        const winScroll =
          document.body.scrollTop || document.documentElement.scrollTop;
        const height =
          document.documentElement.scrollHeight -
          document.documentElement.clientHeight;
        const scrolled = (winScroll / height) * 100;
        document.getElementById("scrollIndicator").style.width = scrolled + "%";
        // Update active nav link
        const sections = document.querySelectorAll("section");
        const navLinks = document.querySelectorAll(".nav-link");
        let current = "";
        sections.forEach((section) => {
          const sectionTop = section.offsetTop;
          const sectionHeight = section.clientHeight;
          if (pageYOffset >= sectionTop - 200) {
            current = section.getAttribute("id");
          }
        });
        navLinks.forEach((link) => {
          link.classList.remove("active-nav");
          if (link.getAttribute("href").substring(1) === current) {
            link.classList.add("active-nav");
          }
        });
      };

      // Animate skill meters on scroll
      function animateSkillMeters() {
        const skillMeters = document.querySelectorAll(".skill-meter");
        skillMeters.forEach((meter) => {
          const width = meter.getAttribute("data-width");
          meter.style.width = width;
        });
      }
      // Check if skill section is in view
      function checkSkillsInView() {
        const skillsSection = document.getElementById("skills");
        const skillsSectionTop = skillsSection.offsetTop;
        const skillsSectionHeight = skillsSection.clientHeight;
        if (
          window.pageYOffset > skillsSectionTop - 500 &&
          window.pageYOffset < skillsSectionTop + skillsSectionHeight
        ) {
          animateSkillMeters();
          window.removeEventListener("scroll", checkSkillsInView);
        }
      }
      window.addEventListener("scroll", checkSkillsInView);

          this.querySelector("img").style.transform = "scale(1)";
        });

      const certCards = document.querySelectorAll(".cert-card");
      certCards.forEach((card) => {

            viewDetailsBtn.textContent = "Read More ‚Üí";

            viewDetailsBtn.textContent = "Read More ‚Üí";

      function createParticles() {
        const contactSection = document.getElementById("contact");
        if (!contactSection) return;
        for (let i = 0; i < 10; i++) {
          const particle = document.createElement("div");
          particle.classList.add("particle");
          const size = Math.random() * 8 + 2;
          const posX = Math.random() * 100;
          const posY = Math.random() * 100;
          const delay = Math.random() * 5;
          const duration = Math.random() * 10 + 10;
          particle.style.width = `${size}px`;
          particle.style.height = `${size}px`;
          particle.style.left = `${posX}%`;
          particle.style.top = `${posY}%`;
          particle.style.animation = `float ${duration}s ease-in-out ${delay}s infinite`;
          contactSection.appendChild(particle);
        }
