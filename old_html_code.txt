    // View More Functionality
    function setupViewMore(containerId, buttonId, initialCount) {
      const container = document.getElementById(containerId);
      if (!container) return;
      const items = container.querySelectorAll(`.${containerId.replace('-container', '-item')}`);
      const button = document.getElementById(buttonId);
      let itemsVisible = initialCount;

      items.forEach((item, index) => {
        if (index >= initialCount) {
          item.classList.add("hidden");
        }
      });

      if (items.length <= initialCount) {
        button.classList.add('hidden');
      }

      return function () {
        const isShowingAll = itemsVisible === items.length;
        if (isShowingAll) {
          items.forEach((item, index) => {
            if (index >= initialCount) {
              item.classList.add("hidden");
            }
          });
          itemsVisible = initialCount;
          button.innerHTML = `View More ${containerId.includes('gallery') ? 'Gallery Photos' : 'Certifications'} <i data-feather="arrow-down" class="ml-2"></i>`;
          document.getElementById(containerId.replace('-container', '')).scrollIntoView({ behavior: 'smooth', block: 'start' });
        } else {
          items.forEach((item) => {
            item.classList.remove("hidden");
          });
          itemsVisible = items.length;
          button.innerHTML = `View Less <i data-feather="arrow-up" class="ml-2"></i>`;
        }
        feather.replace();
      };
